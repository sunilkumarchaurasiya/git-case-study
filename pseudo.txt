# Pseudocode for a program to calculate the sum of two numbers

# Prompt the user to enter the first number
input "Enter the first number: " as first_number

# Prompt the user to enter the second number
input "Enter the second number: " as second_number

# Calculate the sum of the two numbers
sum = first_number + second_number

# Display the result to the user
output "The sum of" first_number "and" second_number "is" sum
